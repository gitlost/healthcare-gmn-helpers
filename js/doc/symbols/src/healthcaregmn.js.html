1 /**
  2  * Helper module that is both a demonstration and usable implementation of a
  3  * check character generator and verifier for a GS1 GMN that is used for
  4  * Regulated Healthcare medical devices that fall under the EU regulations EU
  5  * MDR 2017/745 and EU IVDR 2017/746, herein referred to as a "healthcare GMN".
  6  *
  7  * @author (c) 2019 GS1 AISBL. All rights reserved
  8  *
  9  * @namespace HealthcareGMN
 10  *
 11  */
 12 var HealthcareGMN = (function () {
 13 
 14     "use strict";
 15 
 16     /**
 17      * Descending primes used as multipliers of each data character.
 18      * @private
 19      */
 20     var weights =
 21         [ 83,79,73,71,67,61,59,53,47,43,41,37,31,29,23,19,17,13,11,7,5,3,2 ];
 22 
 23     /**
 24      * GS1 AI encodable character set 82. Place in the string represents the
 25      * character value.
 26      * @private
 27      */
 28     var cset82 =
 29         "!\"%&'()*+,-./0123456789:;<=>?ABCDEFGHIJKLMNOPQRSTUVWXYZ" +
 30         "_abcdefghijklmnopqrstuvwxyz";
 31 
 32     /**
 33      * Subset of the encodable character set used for check characters.
 34      * @private
 35      */
 36     var cset32 = "23456789ABCDEFGHJKLMNPQRSTUVWXYZ";
 37 
 38     /**
 39      * Character to value map for cset82.
 40      * @private
 41      */
 42     var cset82value = {};
 43     for (var i = 0; i < cset82.length; i++)
 44         cset82value[ cset82[i] ] = i;
 45 
 46     /**
 47      * Calculates the two check characters for a given partial healthcare GMN.
 48      *
 49      * @memberof HealthcareGMN
 50      * @param {string} part a partial healthcare GMN.
 51      * @return {string} two check characters.
 52      * @throws Will throw error if the format of the given healthcare GMN is invalid.
 53      */
 54     var checkCharacters = function (part)
 55     {
 56         _formatChecks(part, false);
 57 
 58         /*
 59          * The GMN check digit calculation is performed here.
 60          *
 61          */
 62 
 63         // Characters are compared with the rightmost weights
 64         var offset = 23 - part.length;
 65 
 66         // Modulo 1021 sum of the products of the character values and their
 67         // corresponding weights
 68         var sum = 0;
 69         for (var i = 0; i < part.length; i++)
 70         {
 71                 var c = cset82value[ part[i] ];
 72                 var w = weights[ offset + i ];
 73                 sum += c * w;
 74         }
 75         sum %= 1021;
 76 
 77         // Split the 10-bit sum over two five-bit check characters
 78         return "" + cset32[ Math.floor(sum / 32) ] + cset32[ sum % 32 ];
 79     };
 80 
 81     /**
 82      * Complete a given partial healthcare GMN by appending two check characters.
 83      *
 84      * @memberof HealthcareGMN
 85      * @param {string} part a partial healthcare GMN.
 86      * @return {string} a complete healthcare GMN including check characters.
 87      * @throws Will throw error if the format of the given healthcare GMN is invalid.
 88      */
 89     var addCheckCharacters = function (part)
 90     {
 91         return part + checkCharacters(part);
 92     };
 93 
 94     /**
 95      * Verify that a given healthcare GMN has correct check characters.
 96      *
 97      * @memberof HealthcareGMN
 98      * @param {string} gmn a healthcare GMN.
 99      * @return {boolean} true if the healthcare GMN is has valid check characters. Otherwise false.
100      * @throws Will throw error if the format of the given healthcare GMN is invalid.
101      */
102     var verifyCheckCharacters = function (gmn)
103     {
104         _formatChecks(gmn, true);
105 
106         // Split off the provided check characters, recalculate them and ensure
107         // that they match
108         var part = gmn.substring(0, gmn.length - 2);
109         var suppliedChecks = gmn.substring(gmn.length - 2, gmn.length);
110 
111         return checkCharacters(part) === suppliedChecks;
112     };
113 
114     // Perform some local consistency checks on the input
115     var _formatChecks = function (input, complete) {
116         var maxLength = complete ? 25 : 23;
117         var minLength = complete ? 8 : 6;
118 
119         // Verify length
120         if (input.length < minLength)
121             throw "The input is too short.";
122         if (input.length > maxLength)
123             throw "The input is too long.";
124 
125         // Ensure that first five digits are numeric
126         var i;
127         for (i = 0; i < 5; i++)
128         {
129             if ("0123456789".indexOf( input[i] ) == -1)
130                 throw "First five characters must be digits.";
131         }
132 
133         // Verify that the remaining content is in the encodable character set
134         for (i = 5; i < input.length; i++)
135         {
136             if (typeof cset82value[ input[i] ] === "undefined")
137                 throw "Invalid character at position " + (i + 1) + ": " + input[i];
138         }
139 
140     };
141 
142     return {
143         verifyCheckCharacters: verifyCheckCharacters,
144         checkCharacters: checkCharacters,
145         addCheckCharacters: addCheckCharacters,
146     };
147 
148 })();
149 
150 
151 // For node.js
152 try {
153    module.exports = exports = HealthcareGMN;
154 } catch (e) {}
